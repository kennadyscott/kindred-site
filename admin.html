Connect live data (Supabase) — currently counting in this browser only

This static prototype counts events per-browser. To count every visitor, create a free Supabase project and run this SQL — it creates an insert-only events table, so even the database can only receive anonymous counts:

create table events (
  id bigint generated always as identity primary key,
  event text not null,
  props jsonb default '{}'::jsonb,
  created_at timestamptz default now()
);
alter table events enable row level security;
-- visitors may add a count; nobody anonymous may read them back
create policy "insert only" on events for insert to anon with check (true);

-- Founding-therapist leads (business contacts who opt in on the For Therapists page).
-- Insert-only for anon; read them back with your service-role key in a real admin backend.
create table therapist_leads (
  id bigint generated always as identity primary key,
  name text, email text, cred text, state text, plan text,
  ts timestamptz default now()
);
alter table therapist_leads enable row level security;
create policy "insert only" on therapist_leads for insert to anon with check (true);
